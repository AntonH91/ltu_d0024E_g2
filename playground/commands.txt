DOWNLOAD
	git pull	=	Downloads remote repository and merges with local changes
	
	git fetch	=	Downloads remote repository
	
	git merge	=	Merges branches together
	
	git rebase	=	Rebases branches together

BRANCHING
	git branch	=	Create a new branch
	git checkout =	Check out a different branch (or reset a file to the committed version, if given a file/folder name)
	git stash	=	Stash uncommitted changes for migrating between branches
	git stash apply = Put stashed changes in the checked out branch

CHECK
	git status	=	Check status of local repository
	git log		=	Check commit log
	git diff	=	Check changes
UPLOAD

	git add 	=	stage changes 
	
	git commit	=	add changes to index 

	git push	= 	publish changes - push changes to remote repository
